Usage
1.Compute BB84 link capacities
from capacity_model.bb84_capacity import compute_capacity_matrix
C = compute_capacity_matrix(node_positions, pulse_rate=1e9)

2.Compute widest-path capacities
from routing.widest_path import widest_path_all_pairs
C_star = widest_path_all_pairs(C)

3.Run an adaptive attack simulation
from simulation.simulate_robustness import run_attack

curve = run_attack(
    C_initial=C,
    topology=G,
    strategy="betweenness_weighted"
)
